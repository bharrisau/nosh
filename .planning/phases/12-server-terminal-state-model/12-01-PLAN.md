```rust
impl CellStyle {
    pub const NONE: u8 = 0x00;
    pub const BOLD: u8 = 0x01;
    pub const ITALIC: u8 = 0x02;
    pub const UNDERLINE: u8 = 0x04;
    pub const REVERSE: u8 = 0x08;
}
pub struct CursorPos { pub row: u16, pub col: u16 }   // 0-based
pub struct DiffRun { pub row: u16, pub start_col: u16, pub style: CellStyle, pub fg: u8, pub bg: u8, pub chars: String }
```
NOTE: `fg`/`bg` are `u8` (NOT Option) — value `0` = terminal-default color. The Cell type
must preserve the default-vs-explicit distinction by treating `0` as "default" and never
emitting an explicit `0` for a default cell (do not collapse default to a black palette write).
